Class Phalcon\Binary\Reader¶

Source on GitHub

Provides utilities to work with binary data

<?php

$fp = fopen('unit-tests/assets/data.bin', 'rb');
$bin = new Phalcon\Binary\Reader($fp);
$v = $bin->readUnsignedChar();
