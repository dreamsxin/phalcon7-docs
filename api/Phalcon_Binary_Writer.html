Class Phalcon\Binary\Writer¶

Source on GitHub

Provides utilities to work with binary data

<?php

$fp = fopen('unit-tests/assets/data.bin', 'wb');
$bin = new Phalcon\Binary\Writer($fp);
$bin->writeUnsignedChar(1);
