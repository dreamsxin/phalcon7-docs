Class Phalcon\Debug\Dump¶

Source on GitHub

Dumps information about a variable(s)

<?php

    $foo = 123;
    echo (new \Phalcon\Debug\Dump())->variable($foo, "foo");

<?php

    $foo = "string";
    $bar = ["key" => "value"];
    $baz = new stdClass();
    echo (new \Phalcon\Debug\Dump())->variables($foo, $bar, $baz);
