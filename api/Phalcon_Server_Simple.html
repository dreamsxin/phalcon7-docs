Class Phalcon\Server\Simple¶

Source on GitHub

<?php

 class App extends Phalcon\Application {
     public function handle($data = NULL) {
         $data = trim($data);
         if (empty($data)) {
             return 'Please input'.PHP_EOL;
         } else {
             return '>>> '.$data.PHP_EOL;
         }
    }
 }
 $server = new Phalcon\Server\Simple();
 $server->start(new App);
